reg <- function (model, title = '', ptest = TRUE) {
  test <- coeftest(model, vcovHC(model))
  if(ptest) {
    print(test)
    stargazer(test[,], title = title, type='text')
  }
  return (list('model' = model, 'test' = test))
}
